def plot_data(file_name,title):
    y = []
    with open(file_name) as f:
        for line in f:
            temp = line.rstrip(os.linesep)
            y.append(float(temp))
        f.close()
    ax = sns.distplot(y)
    ax.set_xlabel('logBB')
    ax.set_ylabel('Probability Density')
    ax.set_title(title)
    return y
